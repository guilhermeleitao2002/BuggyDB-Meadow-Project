def eh_animal(obj):
    # universal -> booleano
    '''Recebe um qualquer argumento e retorna True se e s se este corresponder a um animal.'''
    return type(obj) == list and len(obj) >= 3 and type(obj[0]) == str and len(obj[0]) > 0 and\
           type(obj[1]) == int and obj[1] >= 0 and type(obj[2]) == int and obj[2] >= 0 and (len(obj) == 3 or\
           (len(obj) == 5 and type(obj[3]) == int and obj[3] >= 0 and type(obj[4]) == int and obj[4] >= 0))
def eh_predador(obj):
    # universal -> booleano
    '''Recebe um qualquer argumento e retorna True se e s se este corresponder a um animal predador.'''
    return eh_animal(obj) and len(obj) == 5
def eh_presa(obj):
    # universal -> booleano
    '''Recebe um qualquer argumento e retorna True se e s se este corresponder a um animal presa.'''
    return eh_animal(obj) and len(obj) == 3
def animais_iguais(obj1, obj2):
    # animal x animal -> booleano
    '''Recebe dois animais e retorna True se e s se eles forem animais iguais.'''
    return eh_animal(obj1) and eh_animal(obj2) and obj1[0] == obj2[0] and obj1[1] == obj2[1] and\
           obj1[2] == obj2[2] and (len(obj1) == 3 or obj1[3] == obj2[3] and obj1[4] == obj2[4])
def animal_para_char(animal):
    # animal -> str
    '''Recebe um animal e retorna o primeiro carcter do nome da espcie em minsculo, caso se trate de uma presa, ou em maiculo, um predador.'''
    return animal[0][0].lower() if len(animal) == 3 else animal[0][0].upper()
def animal_para_str(animal):
    # animal -> str
    '''Recebe um animal e retorna uma cadeia de caracteres com a devida representao das suas instncias.'''
    return "{} [{}/{}]".format(animal[0], animal[1], animal[2]) if len(animal) == 3 else\
           "{} [{}/{};{}/{}]".format(animal[0], animal[1], animal[2], animal[3], animal[4])
def eh_animal_fertil(animal):
    # animal -> boolenao
    '''Recebe um animal e retorna True se e s se o animal atingiu a idade de reproduo.'''
    return obter_freq_reproducao(animal) <= obter_idade(animal)
def eh_animal_faminto(animal):
    # animal -> booleano
    '''Recebe um animal e retorna True se e s se se trata de um predador e o nivel de fome do animal atingiu ou superou a frequncia de alimentao.'''
    return eh_predador(animal) and obter_freq_alimentacao(animal) <= obter_fome(animal)
def reproduz_animal(animal):
    # animal -> animal
    '''Recebe um animal e altera a sua idade para zero, retornando um animal da mesma espcie, com os mesmos valores de frequncias, mas com a idade e fome,
       no caso de predador, a zeros.'''
    reset_idade(animal)
    return cria_animal(obter_especie(animal), obter_freq_reproducao(animal), obter_freq_alimentacao(animal))

def cria_prado(d, r, a, p):
    # posicao x tuplo x tuplo x tuplo -> prado
    '''Recebe uma posio correspondente  montanha do canto inferior direito, proporcionando as dimenses do prado, um tuplo com zero ou mais
       posies de rochedos, um tuplo com um ou mais animais e um outro tuplo com as respetivas posies dos animais anteriores e retorna
       um prado com todos os atributos passados.'''
    if not eh_posicao(d) or not eh_posicao(r) and not eh_tuplo_posicoes(r) and len(r) != 0 or not eh_animal(a) and not eh_tuplo_animais(a)\
       or not eh_posicao(p) and not eh_tuplo_posicoes(p) or eh_animal(a) and not eh_posicao(p) or not eh_animal(a) and eh_posicao(p) or eh_tuplo_animais(a) and eh_tuplo_posicoes(p) and len(a) != len(p):
        raise ValueError("cria_prado: argumentos invalidos")
    meadow = [[[] for x in range(obter_pos_x(d) - 1)] for y in range(obter_pos_y(d) - 1)] # criar um prado ignorando as montanhas  volta (simplificar)
    if eh_posicao(r): # se houver apenas um rochedo
        meadow[obter_pos_y(r) - 1][obter_pos_x(r) - 1] = "@"
    else: # se eventualmente houver vrios rochedos
        for pos in r:
            meadow[obter_pos_y(pos) - 1][obter_pos_x(pos) - 1] = "@"
    if eh_animal(a) and eh_posicao(p): # se houver apenas um animal e respetiva posio
        meadow[obter_pos_y(p) - 1][obter_pos_x(p) - 1] = cria_copia_animal(a)
    else: # se eventualmente houver vrios animais
        for i in range(len(a)):
            meadow[obter_pos_y(p[i]) - 1][obter_pos_x(p[i]) - 1] = cria_copia_animal(a[i])
    return meadow
def eh_tuplo_posicoes(t):
    # tuplo -> booleano
    '''Funo auxiliar que recebe um tuplo e retorna True se e s se todos os seus elementos forem posies.'''
    for p in t:
        if not eh_posicao(p):
            return False
    return True
def eh_tuplo_animais(t):
    # tuplo -> booleano
    '''Funo auxiliar que recebe um tuplo e retorna True se e s se todos os seus elementos forem animais.'''
    for a in t:
        if not eh_animal(a):
            return False
    return True
def cria_copia_prado(meadow):
    # prado -> prado
    '''Recebe um qualquer argumento e, caso este seja um prado, retorna uma cpia exata do prado passado.'''
    if not eh_prado(meadow):
        raise ValueError("cria_prado: argumentos invalidos")
    cp = [[[] for x in range(obter_tamanho_x(meadow) - 2)] for y in range(obter_tamanho_y(meadow) - 2)]
    for i in range(obter_tamanho_y(meadow) - 2):
        for j in range(obter_tamanho_x(meadow) - 2):
            if eh_animal(meadow[i][j]):
                cp[i][j] = cria_copia_animal(meadow[i][j])
            else:
                cp[i][j] = meadow[i][j]
    return cp
def obter_tamanho_x(meadow):
    # prado -> int
    '''Recebe um prado e retorna o nmero de colunas (tem em conta as montanhas  volta).'''
    return len(meadow[0]) + 2
def obter_tamanho_y(meadow):
    # prado -> int
    '''Recebe um prado e retorna o nmero de linhas (tem em conta as montanhas  volta).'''
    return len(meadow) + 2
def obter_numero_predadores(meadow):
    # prado -> int
    '''Recebe um prado e devolve o nmero de animais predadores existentes nele.'''
    num = 0
    for i in range(len(meadow)):
        for j in range(len(meadow[0])):
            if eh_predador(meadow[i][j]):
                num += 1
    return num
def obter_numero_presas(meadow):
    # prado -> int
    '''Recebe um prado e devolve o nmero de animais presas existentes nele.'''
    num = 0
    for i in range(len(meadow)):
        for j in range(len(meadow[0])):
            if eh_presa(meadow[i][j]):
                num += 1
    return num
def obter_posicao_animais(meadow):
    # prado -> tuplo posicoes
    '''Recebe um prado e devolve um tuplo de todas as posies onde se encontram animais nele.'''
    pos = ()
    for i in range(len(meadow)):
        for j in range(len(meadow[0])):
            if eh_animal(meadow[i][j]):
                pos += (cria_posicao(j + 1, i + 1),)
    return pos
def obter_animal(meadow, pos):
    # prado x posicao -> animal
    '''Recebe um prado e uma posio e devolve o animal que se encontra nela.'''
    return meadow[obter_pos_y(pos) - 1][obter_pos_x(pos) - 1]
def eliminar_animal(meadow, pos):
    # prado x posicao -> prado
    '''Recebe um prado e uma posio, modifica-o eliminando o animal da respetiva posio e retorna o mesmo prado.'''
    meadow[obter_pos_y(pos) - 1][obter_pos_x(pos) - 1] = []
    return meadow
def mover_animal(meadow, pos1, pos2):
    # prado x posicao x posicao -> prado
    '''Recebe um prado e duas posies, modifica-o movendo o animal da primeira posio para a segunda e retorna o mesmo prado.'''
    meadow[obter_pos_y(pos2) - 1][obter_pos_x(pos2) - 1] = obter_animal(meadow, pos1)
    eliminar_animal(meadow, pos1)
    return meadow
def inserir_animal(meadow, animal, pos):
    # prado x animal x posicao -> prado
    '''Recebe um prado, um animal e uma posio e introduz destrutivamente no prado o animal na posio indicada, retornando o mesmo prado.'''
    meadow[obter_pos_y(pos) - 1][obter_pos_x(pos) - 1] = animal
    return meadow
def eh_prado(obj):
    # universal -> booleano
    '''Recebe um qualquer argumento e retorna True se e s se este for um prado.'''
    if type(obj) != list or len(obj) < 3:
        return False
    for y in range(obter_tamanho_y(obj) - 2):
        if type(obj[y]) != list or len(obj[y]) < 3:
            return False
        for x in range(obter_tamanho_x(obj) - 2):
            if obj[y][x] != [] and obj[y][x] != "@" and not eh_animal(obj[y][x]):
                return False
    return True
def eh_posicao_animal(meadow, pos):
    # prado x posicao -> booleano
    '''Recebe um prado e uma posio e retorna True se e s se nessa posio se encontrar um animal.'''
    if obter_pos_y(pos) == 0 or obter_pos_x(pos) == 0 or obter_pos_y(pos) == len(meadow) + 1 or obter_pos_x(pos) == len(meadow[0]) + 1:
        return False
    return eh_animal(meadow[obter_pos_y(pos) - 1][obter_pos_x(pos) - 1])
def eh_posicao_obstaculo(meadow, pos):
    # prado x posicao -> booleano
    '''Recebe um prado e uma posio e retorna True se e s se nessa posio se encontrar um rochedo ou montanha.'''
    return obter_pos_y(pos) == 0 or obter_pos_x(pos) == 0 or obter_pos_y(pos) == len(meadow) + 1\
           or obter_pos_x(pos) == len(meadow[0]) + 1 or meadow[obter_pos_y(pos) - 1][obter_pos_x(pos) - 1] == "@"
def eh_posicao_livre(meadow, pos):
    # prado x posicao -> booleano
    '''Recebe um prado e uma posio e retorna True se e s se essa posio se encontrar livre.'''
    if obter_pos_y(pos) == 0 or obter_pos_x(pos) == 0 or obter_pos_y(pos) == len(meadow) + 1 or obter_pos_x(pos) == len(meadow[0]) + 1:
        return False
    return meadow[obter_pos_y(pos) - 1][obter_pos_x(pos) - 1] == []
def prados_iguais(meadow1, meadow2):
    # prado x prado -> booleano
    '''Recebe dois quaisquer argumentos e retorna True se e s se ambos forem prados e tiverem exatamente os mesmos atributos.'''
    if not eh_prado(meadow1) or not eh_prado(meadow2) or len(meadow1) != len(meadow2):
        return False
    for y in range(len(meadow1)):
        for x in range(len(meadow1[0])):
            if meadow1[y][x] != meadow2[y][x] and not animais_iguais(meadow1[y][x], meadow2[y][x]):
                return False
    return True
def prado_para_str(meadow):
    # prado -> str
    '''Recebe um prado e retorna uma cadeia de caracteres com a respetiva representao espacial do mesmo.'''
    # desenhar a primeira linha
    m = "+"
    for i in range(len(meadow[0])): m += "-"
    m += "+\n"
    # desenhar as linhas intermdias tendo em conta o que representam
    for i in range(len(meadow)):
        m += "|"
        for e in meadow[i]:
            if eh_animal(e): #  uma animal
                m += animal_para_char(e)
            elif e == "@": #  um rochedo
                m += e
            else: #  espao livre
                m += "."
        m += "|\n"
    # desenhar a ltima linha
    m += "+"
    for i in range(len(meadow[0])): m += "-"
    return m + "+"
def obter_valor_numerico(meadow, pos):
    # prado x posicao -> int
    '''Recebe um prado e uma posio e devolve o valor numrico da posio correspondente nesse prado.'''
    return obter_pos_y(pos) * (obter_tamanho_x(meadow)) + obter_pos_x(pos)
def obter_movimento(meadow, pos):
    # prado x posicao -> posicao
    '''Recebe um prado e uma posio e devolve a posio que resulta da aplicao das regras de movimento definidas no enunciado.'''
    animal = obter_animal(meadow, pos)
    t_pos2 = obter_posicoes_adjacentes(pos) # fazer uso da funo anteriormente definida
    t_pos = ()
    # ignorar as posies com obstculos, respeitando os casos dos diferentes tipos de animais
    if eh_predador(animal):
        for p in t_pos2:
            if not eh_posicao_obstaculo(meadow, p) and eh_presa(obter_animal(meadow, p)):
                t_pos += (cria_copia_posicao(p),)
        if len(t_pos) == 0:
            for p in t_pos2:
                if not eh_posicao_obstaculo(meadow, p):
                    t_pos += (cria_copia_posicao(p),)
    else:
        for p in t_pos2:
            if not eh_posicao_obstaculo(meadow, p) and eh_posicao_livre(meadow, p):
                t_pos += (cria_copia_posicao(p),)
    for i in range(len(t_pos)):
        if obter_valor_numerico(meadow, pos) % len(t_pos) == i: # N (mod p)
            return cria_copia_posicao(t_pos[i])
    return cria_copia_posicao(pos)

def geracao(meadow):
    # prado -> prado
    '''Recebe um prado e retorna-o modificado, simulando a passagem de uma gerao completa, conforme descrita no enunciado.'''
    animals_p = obter_posicao_animais(meadow) # obter a posio de todos os animais em vez de perder tempo iterando pelo prado inteiro
    black_list = ()
    for animal_p in animals_p:
        skip, fed = False, False
        for p in black_list:
            if posicoes_iguais(p, animal_p):
                skip = True
        if skip:
            continue
        animal = obter_animal(meadow, animal_p)
        animal = aumenta_idade(animal) # incrementa idade
        animal = aumenta_fome(animal) # incrementa fome
        final_p = obter_movimento(meadow, animal_p) # obter o movimento de eleio para cada animal de acordo com as suas caractersticas
        if not posicoes_iguais(final_p, animal_p):
            if eh_predador(animal) and eh_presa(obter_animal(meadow, final_p)): # se for um animal predador tem de se preocupar com o seu nivel de fome
                fed = True
                black_list += (cria_copia_posicao(final_p),)
            meadow = mover_animal(meadow, animal_p, final_p)
            if eh_animal_fertil(animal):
                meadow = inserir_animal(meadow, reproduz_animal(animal), animal_p)
            if fed:
                animal = reset_fome(animal)
            if eh_animal_faminto(animal):
                meadow = eliminar_animal(meadow, final_p)
        else:
            if eh_animal_faminto(animal):
                meadow = eliminar_animal(meadow, animal_p)
    return meadow
